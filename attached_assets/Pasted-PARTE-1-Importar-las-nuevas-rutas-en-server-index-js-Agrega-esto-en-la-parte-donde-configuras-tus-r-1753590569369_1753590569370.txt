import { useEffect, useState } from "react";
import axios from "axios";

export default function Dashboard() {
  const [data, setData] = useState({ totalLeads: 0, totalClients: 0, totalQuotations: 0 });
  const [error, setError] = useState("");

  useEffect(() => {
    const token = localStorage.getItem("token");

    axios.get("/api/analytics/summary", {
      headers: { Authorization: `Bearer ${token}` }
    })
    .then(res => setData(res.data))
    .catch(err => {
      console.error(err);
      setError("Error al cargar el dashboard");
    });
  }, []);

  return (
    <div className="min-h-screen bg-gray-100 p-6">
      <h1 className="text-3xl font-bold mb-6">Panel General</h1>
      {error && <p className="text-red-600">{error}</p>}
      
      <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
        <Card title="Total de Leads" value={data.totalLeads} />
        <Card title="Clientes registrados" value={data.totalClients} />
        <Card title="Cotizaciones emitidas" value={data.totalQuotations} />
      </div>
    </div>
  );
}

function Card({ title, value }) {
  return (
    <div className="bg-white shadow p-6 rounded">
      <h2 className="text-xl font-semibold mb-2">{title}</h2>
      <p className="text-4xl font-bold text-blue-600">{value}</p>
    </div>
  );
}
